users[icon: user, color: Green] {
  user_id serial pk
  name string
  email string unique
  role enum ["trainer", "client"]
}

trainers[icon: activity, color: Green] {
  trainer_id int pk fk  
  specialization string
  experience int
}

clients[icon: users, color: Green] {
  client_id int pk fk
  age int
  gender string
  fitness_goal string
}

plans[icon: package, color: Green, colorMode: bold] {
  plan_id serial pk
  trainer_id int fk
  name string
  description string
  price float
  duration_days int
  type enum ["consultation", "coaching"]
}


subscriptions[icon: repeat, color: green] {
  subscription_id serial pk
  client_id int fk
  plan_id int fk
  trainer_id int fk 
  start_date datetime
  end_date datetime
  status enum ["active", "completed", "cancelled"]
}


sessions[icon: calendar, color: Green] {
  session_id serial pk
  trainer_id int fk
  client_id int fk
  subscription_id int fk nullable
  scheduled_at datetime
  type enum ["consultation", "live_session"]
  status enum ["scheduled", "completed", "cancelled"]
}


checkins[icon: clipboard, color: Green] {
  checkin_id serial pk
  client_id int fk
  subscription_id int fk
  date datetime
  notes string
}


progress_logs[icon: trending-up, color: Green] {
  progress_id serial pk
  checkin_id int fk  
  weight float
  body_fat float
  measurements string
}


trainer_notes[icon: edit, color: Green] {
  note_id serial pk
  trainer_id int fk
  entity_type enum ["checkin", "session"]  
  entity_id int
  content string
}


payments[icon: credit-card, color: Green] {
  payment_id serial pk
  subscription_id int fk
  amount float
  method enum ["upi", "card"]
  status enum ["pending", "completed"]
  transaction_type enum ["payment", "refund"]   
  paid_at datetime
}
